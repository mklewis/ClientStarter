Responsive ng-table example

Implementation of ng-table using ngTableDynamic directive. We've added the responsive option to collapse columns into detail table rows. Resize the screen to see the columns get removed for the table row and added to an accordian style detail list

{{item[col.field]}} 
   <table ng-table-dynamic="ngtableCtrl.tableParams with ngtableCtrl.headings" 
        class="table" show-filter="ngtableCtrl.showFilters">
        <tbody ng-repeat="item in $data">
            <tr ng-click="item.showDetailAttr = !item.showDetailAttr">
                <td ng-repeat="col in $columns">
                    <span ng-if="$index == 0 && ngtableCtrl.detailsToShow()" 
                        class="fa" ng-class="{'fa-caret-right': !item.showDetailAttr, 'fa-caret-down': item.showDetailAttr}">
                    </span>
                    {{item[col.field]}}
                </td>
            </tr>
            <tr ng-show="item.showDetailAttr && ngtableCtrl.detailsToShow()" 
                detail-row item-object="item" table-ctrl="ngtableCtrl" headings="headings" show-column-fn="showColumn">
            </tr>
        </tbody>
    </table>
